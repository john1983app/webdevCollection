function CalculateTotalFund(){ var iOrigFund = document.getElementById("originfund"); var fRat = document.getElementById("rates"); var iiv = document.getElementById("interval"); var TotalFund,i,a = 0; var Rate = parseFloat(fRat.value); var OriginFund = parseFloat(iOrigFund.value); var Interval =parseInt(iiv.value); for(i=Interval;i>0;i--){ a += Math.pow(Rate+1,i); } TotalFund = OriginFund * a; document.write("你的本金是:"+OriginFund); document.write("你的利率是:"+ Rate); document.write("你的本利和是:"+TotalFund); }
